CREATE DATABASE openai CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

create table api_keys
(
    id          int auto_increment
        primary key,
    api_key     varchar(255)                           not null,
    user_id     int                                    not null,
    name        varchar(100) default 'Default Key'     null,
    is_active   tinyint(1)   default 1                 null,
    rate_limit  int          default 100               null,
    daily_limit int          default 1000              null,
    used_count  int          default 0                 null,
    expires_at  timestamp                              null,
    created_at  timestamp    default CURRENT_TIMESTAMP null,
    updated_at  timestamp    default CURRENT_TIMESTAMP null on update CURRENT_TIMESTAMP,
    constraint api_key
        unique (api_key),
    constraint api_keys_ibfk_1
        foreign key (user_id) references users (id)
            on delete cascade
);

create index idx_api_key
    on api_keys (api_key);

create index idx_created_at
    on api_keys (created_at);

create index idx_user_id
    on api_keys (user_id);



create table completions
(
    id                 int auto_increment
        primary key,
    completion_id      varchar(64)                                                                                  not null,
    user_id            int                                                                                          not null,
    api_key_id         int                                                                                          not null,
    model              varchar(100)                                                                                 not null,
    request_messages   json                                                                                         not null,
    response_content   text                                                                                         null,
    status             enum ('pending', 'processing', 'completed', 'failed', 'cancelled') default 'pending'         null,
    temperature        decimal(3, 2)                                                      default 1.00              null,
    stream             tinyint(1)                                                         default 0                 null,
    finish_reason      varchar(20)                                                        default 'stop'            null,
    usage_tokens       int                                                                default 0                 null,
    processing_time_ms int                                                                default 0                 null,
    error_message      text                                                                                         null,
    started_at         timestamp                                                                                    null,
    completed_at       timestamp                                                                                    null,
    created_at         timestamp                                                          default CURRENT_TIMESTAMP null,
    updated_at         timestamp                                                          default CURRENT_TIMESTAMP null on update CURRENT_TIMESTAMP,
    constraint completion_id
        unique (completion_id),
    constraint completions_ibfk_1
        foreign key (user_id) references users (id)
            on delete cascade,
    constraint completions_ibfk_2
        foreign key (api_key_id) references api_keys (id)
            on delete cascade
);

create index api_key_id
    on completions (api_key_id);

create index idx_completion_id
    on completions (completion_id);

create index idx_created_at
    on completions (created_at);

create index idx_status
    on completions (status);

create index idx_user_created
    on completions (user_id, created_at);

create index idx_user_id
    on completions (user_id);


create table models
(
    id                 bigint auto_increment
        primary key,
    model_name         varchar(100)                         not null,
    display_name       varchar(100)                         not null,
    description        text                                 null,
    owner_id           bigint                               null,
    is_active          tinyint(1) default 1                 null,
    max_tokens         int        default 4096              null,
    supports_streaming tinyint(1) default 1                 null,
    created_at         timestamp  default CURRENT_TIMESTAMP null,
    updated_at         timestamp  default CURRENT_TIMESTAMP null on update CURRENT_TIMESTAMP,
    constraint model_name
        unique (model_name)
);

create index idx_is_active
    on models (is_active);

create index idx_model_name
    on models (model_name);

create index owner_id
    on models (owner_id);

create table users
(
    id            int auto_increment
        primary key,
    username      varchar(50)                          not null,
    email         varchar(100)                         not null,
    password_hash varchar(255)                         not null,
    is_active     tinyint(1) default 1                 null,
    created_at    timestamp  default CURRENT_TIMESTAMP null,
    updated_at    timestamp  default CURRENT_TIMESTAMP null on update CURRENT_TIMESTAMP,
    constraint email
        unique (email),
    constraint username
        unique (username)
);

create table files
(
    id             int auto_increment
        primary key,
    file_id        varchar(64)                                                       not null,
    user_id        int                                                               not null,
    filename       varchar(255)                                                      not null,
    purpose        varchar(50)                                                       not null,
    bytes          bigint                                                            not null,
    status         enum ('uploaded', 'processed', 'error') default 'uploaded'        null,
    status_details text                                                              null,
    created_at     timestamp                               default CURRENT_TIMESTAMP null,
    updated_at     timestamp                               default CURRENT_TIMESTAMP null on update CURRENT_TIMESTAMP,
    constraint file_id
        unique (file_id),
    constraint files_ibfk_1
        foreign key (user_id) references users (id)
            on delete cascade
);

create index idx_file_id
    on files (file_id);

create index idx_purpose
    on files (purpose);

create index idx_user_id
    on files (user_id);

create index idx_email
    on users (email);

create index idx_username
    on users (username);
